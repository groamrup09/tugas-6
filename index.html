Multiple Inheritance dengan Mixin

        * Dart menggunakan keyword Mixins untuk menggunakan beberapa class yang
          berbeda dalam satu class, tanpa perlu membuat inheritance yang rumit.
        * beberapa bahasa yang mendukung multiple inheritance adalah C++, Python,
          Scala, Perl, DLL sedangkan Java, C#, dan Kotlin mendukung Single inheritance
        * Multiple inheritance (pewarisan ganda/banyak) digunakan untuk
          menyelesaikan masalah di mana sebuah objek (subclass) memiliki perilaku atau
          properti dari dua atau lebih induk yang berbeda (superclass) secara bersamaan,
          menggabungkan fungsionalitas tanpa redundansi kode. Contohnya adalah
          pemodelan objek kompleks seperti "Bebek" (Mewarisi bisa berenang dan bisa
          terbang) atau "Smartphone" (Mewarisi Kamera dan Telepon).

mixin CanFly {
  bool _isFlying = false;
  bool get isFlying => _isFlying;
  
  void fly() {
    if (!_isFlying) {
      _isFlying = true;
      print('Started flying!');
    } else {
      print('Already flying!');
    }
  }
  
  void land() {
    if (_isFlying) {
      _isFlying = false;
      print('Landed!');
    } else {
      print('Already on the ground!');
    }
  }
}

mixin CanSwim {
  bool _isSwimming = false;
  bool get isSwimming => _isSwimming;
  
  void swim() {
    if (!_isSwimming) {
      _isSwimming = true;
      print('Started swimming!');
    } else {
      print('Already swimming!');
    }
  }
  
  void stopSwim() {
    if (_isSwimming) {
      _isSwimming = false;
      print('Stop swimming!');
    } else {
      print('Already stopped swimming!');
    }
  }
}

Apa itu? Keyword untuk mengontrol bagaimana class dapat digunakan
Tujuan:
i. Kontrol inheritance/implementasi
ii. Jaminan keamanan tipe
iii. Desain API yang lebih ekspresif

final , 
interface (Dart 3.0+)

        Tidak bisa di-extend atau di-implement dari luar library
        Membatasi hierarki tipe yang diketahui saat compile-time
        Perfect untuk pattern matching (switch exhaustive)

sealed class Status {
  const Status();
}

class Loading extends Status {
  const Loading();
}

class Loaded extends Status {
  final String data;
  const Loaded(this.data);
}

class Failed extends Status {
  final String error;
  const Failed(this.error);
}

void handleStatus(Status status) {
  switch (status) {
    case Loading():
      print('Loading...');
      break;
    case Loaded(:final data):
      print('Data: $data');
      break;
    case Failed(:final error):
      print('Error: $error');
      break;
  }
}

void main() {
  handleStatus(const Loading());
  handleStatus(const Loaded("Data berhasil"));
  handleStatus(const Failed("Terjadi error"));
}

class MathUtils {
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Tidak bisa menghitung faktorial bilangan negatif');
    if (n == 0 || n == 1) return 1;
    int hasil = 1;
    for (int i = 2; i <= n; i++) {
      hasil *= i;
    }
    return hasil;
  }
  
  static bool isPrima(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i * i <= n; i += 2) {
      if (n % i == 0) return false;
    }
    return true;
  }
  
  static double roundToDecimal(double value, int decimalPlaces) {
    double faktor = 1;
    for (int i = 0; i < decimalPlaces; i++) {
      faktor *= 10;
    }
    return (value * faktor).round() / faktor;
  }
  
  static int randomInt(int min, int max) {
    return min + (DateTime.now().millisecondsSinceEpoch % (max - min + 1));
  }
  
  static double randomDouble(double min, double max) {
    return min + (DateTime.now().millisecondsSinceEpoch % 10000) / 10000 * (max - min);
  }
  
  static double rataRata(List angka) {
    if (angka.isEmpty) return 0;
    double sum = 0;
    for (var a in angka) {
      sum += a;
    }
    return sum / angka.length;
  }
  
  static double median(List angka) {
    if (angka.isEmpty) return 0;
    List sorted = List.from(angka)..sort();
    int tengah = sorted.length ~/ 2;
    if (sorted.length % 2 == 0) {
      return (sorted[tengah - 1] + sorted[tengah]) / 2;
    } else {
      return sorted[tengah];
    }
  }
}

void main() {
  print('========== SOAL 5 ==========');
  
  print('Factorial 5: ${MathUtils.factorial(5)}');
  print('Factorial 7: ${MathUtils.factorial(7)}');
  print('Apakah 17 prima? ${MathUtils.isPrima(17)}');
  print('Apakah 20 prima? ${MathUtils.isPrima(20)}');
  print('3.14159 ke 2 desimal: ${MathUtils.roundToDecimal(3.14159, 2)}');
  print('Random Int (1-10): ${MathUtils.randomInt(1, 10)}');
  print('Random Double (0-1): ${MathUtils.randomDouble(0, 1)}');
  
  List nilai = [70, 85, 90, 75, 80, 95];
  print('List nilai: $nilai');
  print('Rata-rata: ${MathUtils.rataRata(nilai)}');
  print('Median: ${MathUtils.median(nilai)}');
}

abstract class MediaPlayer {
  String title;
  int duration;
  
  MediaPlayer(this.title, this.duration);
  
  void play();
  void stop();
  void showInfo();
  
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
}

// ==================== INTERFACES ====================
abstract interface class Playable {
  void play();
  void resume();
}

abstract interface class Pausable {
  void pause();
  void resumeFromPause();
}

// ==================== CLASS IMPLEMENTASI ====================
class AudioPlayer implements MediaPlayer, Playable, Pausable {
  @override
  String title;
  
  @override
  int duration;
  
  String formatAudio;
  int bitrate;
  bool isPlaying = false;
  bool isPaused = false;
  
  AudioPlayer({
    required this.title,
    required this.duration,
    required this.formatAudio,
    required this.bitrate,
  });
  
  @override
  void play() {
    if (isPaused) {
      resume();
    } else {
      isPlaying = true;
      isPaused = false;
      print('🎵 Memutar audio: $title ($formatAudio, ${bitrate}kbps)');
    }
  }
  
  @override
  void resume() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('🎵 Melanjutkan audio: $title');
    }
  }
  
  @override
  void stop() {
    isPlaying = false;
    isPaused = false;
    print('⏹️ Menghentikan audio: $title');
  }
  
  @override
  void pause() {
    if (isPlaying) {
      isPlaying = false;
      isPaused = true;
      print('⏸️ Menjeda audio: $title');
    }
  }
  
  @override
  void resumeFromPause() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('▶️ Melanjutkan audio dari jeda: $title');
    }
  }
  
  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📻 AUDIO PLAYER');
    print('═══════════════════════════════');
    print('Judul: $title');
    print('Format: $formatAudio');
    print('Bitrate: ${bitrate}kbps');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }

}

class VideoPlayer implements MediaPlayer, Playable, Pausable {
  @override
  String title;
  
  @override
  int duration;
  
  String resolusi;
  String formatVideo;
  bool isPlaying = false;
  bool isPaused = false;
  int currentTime = 0;
  
  VideoPlayer({
    required this.title,
    required this.duration,
    required this.resolusi,
    required this.formatVideo,
  });
  
  @override
  void play() {
    if (isPaused) {
      resume();
    } else {
      isPlaying = true;
      isPaused = false;
      print('🎬 Memutar video: $title ($resolusi, $formatVideo)');
    }
  }
  
  @override
  void resume() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('🎬 Melanjutkan video: $title');
    }
  }
  
  @override
  void stop() {
    isPlaying = false;
    isPaused = false;
    currentTime = 0;
    print('⏹️ Menghentikan video: $title');
  }
  
  @override
  void pause() {
    if (isPlaying) {
      isPlaying = false;
      isPaused = true;
      print('⏸️ Menjeda video: $title');
    }
  }
  
  @override
  void resumeFromPause() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('▶️ Melanjutkan video dari jeda: $title');
    }
  }
  
  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📺 VIDEO PLAYER');
    print('═══════════════════════════════');
    print('Judul: $title');
    print('Resolusi: $resolusi');
    print('Format: $formatVideo');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
  
  void setQuality(String quality) {
    print('📺 Mengubah kualitas video ke: $quality');
  }
  
  void seekTo(int seconds) {
    if (seconds >= 0 && seconds <= duration) {
      currentTime = seconds;
      int menit = seconds ~/ 60;
      int detik = seconds % 60;
      print('⏩ Lompat ke ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
    } else {
      print('❌ Waktu tidak valid!');
    }
  }
  
  void fullscreen() {
    print('🖥️ Mode layar penuh aktif');
  }
}
